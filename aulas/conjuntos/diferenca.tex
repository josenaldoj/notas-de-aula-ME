\subsection{Diferença}

\begin{definition}[Diferença de Conjuntos]
\label{def:setminus}
A \textdef{diferença} entre os conjuntos $A$ e $B$, denotada por $B\setminus A$, é definida por:
%
	$$ x \in  B \setminus A \text{ se e somente se }  x \in B \text{ mas } x \notin A $$
\end{definition}
\begin{remark}
Em geral, não é verdade que $A \setminus B = B \setminus A$. Além disso, note que $A^C = \U\setminus A$.
\end{remark}
