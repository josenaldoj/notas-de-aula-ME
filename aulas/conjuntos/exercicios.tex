\begin{exercise}
%\label{exe:}
Demonstre que os seguintes itens são equivalentes:
	\begin{enumerate}[a.]
		\item $A \cup B = B$;
		\item $A \subset B$;
		\item $A \cap B = A$;
	\end{enumerate}
	\begin{tip}
		Para tanto, é preciso provar \textbf{a. $\iff$ b.} e \textbf{b. $\iff$ c.}.
		Outra maneira é provar \textbf{a. $\implies$ b.}, \textbf{b. $\implies$ c.} e por fim, \textbf{c. $\implies$ a.}.
	\end{tip}
\end{exercise}
