\section{Definição de Função Real}

\begin{definition}
    Uma função na forma $f : D \subset \R \to \R$ é chamada de
\textdef{função real} (pois seus valores são números reais, isto é, seu
contradomínio é $\R$) de \textdef{variável real} (pois sua variável
independente assume valores reais, isto é, seu domínio é um
subconjunto de $\R$).
\end{definition}
